<script>

    let allData = [];

    let extraIndex = 0;


    async function loadData() {

        // Carica il file CSV locale o remoto

        Papa.parse("eventi.csv", {

            download: true, header: false, skipEmptyLines: true,

            complete: function(results) {

                allData = results.data.slice(1); // Esclude l'intestazione del CSV

                render();

            }

        });

    }


    function render() {

        const tbody = document.getElementById('board-body');

        tbody.innerHTML = '';

        

        // Mostra solo i primi 7 eventi validi

        allData.slice(0, 7).forEach(row => {

            const tr = document.createElement('tr');

            

            // 7 Colonne Fisse (indici CSV 0-6)

            for(let i=0; i<7; i++) {

                const td = document.createElement('td');

                td.textContent = row[i] || "";

                tr.appendChild(td);

            }


            // 2 Colonne a Rotazione (indici CSV 7-16)

            for(let i=0; i<2; i++) {

                const td = document.createElement('td');

                td.className = 'rotating';

                td.textContent = row[7 + extraIndex + i] || "";

                tr.appendChild(td);

            }

            tbody.appendChild(tr);

        });

    }


    // Cambia le colonne extra ogni 7 secondi

    setInterval(() => {

        extraIndex = (extraIndex + 2) % 10;

        render();

    }, 7000);


    // Ricarica il file CSV ogni 60 secondi

    setInterval(loadData, 60000);

    loadData();

</script>
